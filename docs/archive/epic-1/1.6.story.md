```typescript
interface ContentType {
  id: string;
  name: string;
  pluralName: string;
  icon: string;
  fields: Field[];
```

**Field Model Structure:**
```typescript
interface Field {
  id: string;
  name: string;
  type: 'text' | 'richText' | 'image' | 'date' | 'number' | 'boolean' | 'reference';
  required: boolean;
  defaultValue: any;
  validation: ValidationRule[];
  order: number;
}
```

```typescript
interface ContentItem {

  updatedAt: Date;
}
```
